784. 字母大小写全排列

Published: 周五 26 一月 2018
By Mega Hertz

In Leetcode.

深度优先

class Solution:
    def letterCasePermutation(self, S):
        """
        :type S: str
        :rtype: List[str]
        """
        S = list(S)
        res = []
        def dfs(i):
            if i == len(S):
                res.append(''.join(S[::]))
                return
            dfs(i+1)
            if not S[i].isalpha():
                return
            S[i] = chr(ord(S[i]) ^ 32)
            dfs(i+1)
            S[i] = chr(ord(S[i]) ^ 32)

        dfs(0)
        return res

class Solution:
    def letterCasePermutation(self, S):
        """
        :type S: str
        :rtype: List[str]
        """
        output=['']
        for ch in S:
            if ch.isalpha():
                output=[i+j for i in output for j in [ch.lower(), ch.upper()]]
            else:
                output=[i+ch for i in output]
        return output
