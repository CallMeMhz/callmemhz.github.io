class Solution:
    def fullJustify(self, words, maxWidth):
        lines = [[]]
        current_line = 0
        width = 0  # 当前行宽度（包含空格）
        # 生成行
        for word in words:
            # 若超长则另起一行
            if width + len(word) > maxWidth:
                lines.append([word, ])
                current_line += 1
                # + 1 是预分配单词右侧的一个空格，方便判断边界
                width = len(word) + 1
            else:
                lines[current_line].append(word)
                width += len(word) + 1
        # 补空格
        res = []
        # 最后一行单独处理
        for line in lines[:-1]:
            spaces = maxWidth - sum(map(len, line))  # 剩余空白符个数
            space = (len(line) - 1)
            # 加上条件是为了避免除数为 0
            space_width = (spaces // space) if space else 0
            remanent = (spaces % space) if space else 0
            s = ''
            # 对只有一个单词的行单独处理
            for word in line[:-1]:
                s += word + ' ' * space_width
                if remanent:
                    s += ' '
                    remanent -= 1
            s += line[-1]
            if not space:  # 即 space == 0 该行仅有一个单词
                s += ' ' * (maxWidth - len(line[0]))
            res.append(s)
        s = ' '.join(lines[-1])
        s += ' ' * (maxWidth - len(s))
        res.append(s)
        return res

先把单词按行拍好，然后再插入空格
