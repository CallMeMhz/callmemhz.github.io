    class Solution(object):
    def reverseVowels(self, s):
        vowels = 'aeiouAEIOU'
        seq = []
        for c in s:
            if c in vowels:
                seq.append(c)
        res = []
        for c in s:
            if c in vowels:
                res.append(seq.pop())
            else:
                res.append(c)
        return ''.join(res)

        node = []

        mp = {}
        for word in strs:
            key = ''.join(sorted((word)))
            if key not in mp:
                mp[key] = [word,]
            else:
                mp[key].append(word)
        return list(mp.values())
