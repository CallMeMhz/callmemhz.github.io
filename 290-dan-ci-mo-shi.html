290. 单词模式

Published: 周五 26 一月 2018
By Mega Hertz

In Leetcode.

class Solution:
    def wordPattern(self, pattern, str):
        words = str.split()
        if len(pattern) != len(words):
            return False
        mp = {}
        for i in range(len(pattern)):
            if pattern[i] not in mp:
                if words[i] in mp.values():  # 注意这里检查键值对是否设置过
                    return False
                mp[pattern[i]] = words[i]
            elif mp[pattern[i]] != words[i]:
                return False
        return True

class Solution:
    def wordPattern(self, pattern, str):
        words = str.split()
        if len(pattern) != len(words):
            return False
        return len(set(zip(pattern, words))) == len(set(pattern)) == len(set(words))
