15. 三数之和

Published: 周五 26 一月 2018
By Mega Hertz

In Leetcode.

暴力解法(特别慢)

class Solution:
    def threeSum(self, nums):
        nums = sorted(nums) # O(nlogn)
        output = set()
        for k in range(len(nums)):
            target = -nums[k]
            i,j = k+1, len(nums)-1
            while i < j:
                sum_two = nums[i] + nums[j]
                if sum_two < target:
                    i += 1
                elif sum_two > target:
                    j -= 1
                else:
                    output.add((nums[k],nums[i],nums[j]))
                    i += 1
                    j -= 1
        return list(output)

参考两数之和解法

class Solution:
    def threeSum(self, nums):
        dic = {}
        for i in nums:
            if i not in dic:
                dic[i] = 0
            dic[i] += 1

        if 0 in dic and dic[0] > 2:
            res = [[0,0,0]]
        else:
            res = []

        pos = [p for p in dic if p > 0]
        neg = [n for n in dic if n < 0]

        for p in pos:
            for n in neg:
                inverse = -(p + n)
                if inverse in dic:
                    if inverse == p and dic[p] > 1:
                        res.append([n,p,p])
                    elif inverse == n and dic[n] > 1:
                        res.append([n,n,p])
                    #这里的inverse < n or inverse > p很trick，意味着只取一边的，以防止重复
                    elif inverse < n or inverse > p or inverse == 0:
                        res.append([n,p,inverse])
        return res
