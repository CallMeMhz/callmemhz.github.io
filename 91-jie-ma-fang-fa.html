先给出第一种算法, 然后面试官会让你优化, 这时再写出第二种方法

未优化

时间复杂度：O(N) 空间复杂度：O(N)

class Solution:
    def numDecodings(self, s):
        if not s or s[0] == '0':
            return 0
        if len(s) == 1:
            return 1
        dp = [0] * len(s)
        dp[0] = 1
        if 10 <= int(s[:2]) <= 26:
            if s[1] == '0':
                dp[1] = 1
            else:
                dp[1] = 2
        else:
            if s[1] == '0':
                dp[1] = 0
            else:
                dp[1] = 1
        for i in range(2, len(s)):
            if s[i] != '0':
                dp[i] += dp[i-1]
            if 10 <= int(s[i-1:i+1]) <= 26:
                dp[i] += dp[i-2]
        return dp[-1]

优化

空间复杂度降到 O(1)

class Solution:
    def numDecodings(self, s):
        if not s or s[0] == '0':
            return 0
        c1 = c2 = 1  # c2 保存着上一次的 c1
        for i in range(1, len(s)):
            if s[i] == '0':
                c1 = 0
            if s[i-1] == '1' or s[i-1] == '2' and s[i] <= '6':
                c1, c2 = c1 + c2, c1
            else:
                c2 = c1
        return c1
