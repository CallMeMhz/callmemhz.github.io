8. 字符串转整数

Published: 周五 26 一月 2018
By Mega Hertz

In Leetcode.

一般解法

class Solution:
    def myAtoi(self, s):
        """
        :type str: str
        :rtype: int
        """

        s = s.lstrip()

        if len(s) == 0:
            return 0

        flag = 1;
        start_pos = 0
        nums = [str(i) for i in range(10)]
        _int_min = - 2 ** 31
        _int_max = - _int_min - 1

        if s[0] == '-':
            flag = -1
            start_pos = 1
        elif s[0] == '+':
            start_pos = 1
        res = 0
        for c in s[start_pos:]:
            if c in nums:
                res = res*10 + int(c)*flag
                if res < _int_min:
                    return _int_min
                if res > _int_max:
                    return _int_max
            else:
                return res
        return res
