205. 同构字符串

class Solution:
    def isIsomorphic(self, s, t):
        def helper(s1, s2):
            mp = {}
            for i in range(len(s)):
                if s1[i] not in mp:
                    mp[s1[i]] = s2[i]
                elif mp[s1[i]] != s2[i]:
                    return False
            return True
        return helper(s, t) and helper(t, s)

class Solution(object):
    def isIsomorphic(self, s, t):
        return len(set(zip(s,t))) == len(set(s)) == len(set(t))
