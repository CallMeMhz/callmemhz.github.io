16. 最接近的三数之和

Published: 周五 26 一月 2018
By Mega Hertz

In Leetcode.

class Solution(object):
    def threeSumClosest(self, nums, target):
        n, res, diff = len(nums), 0, float('inf')  # float('inf') 是无穷大
        nums.sort()  # 先排序，方便指针按序移动
        for i in range(n):
            if i > 0 and nums[i] == nums[i-1]:  # 跳过相同的数
                continue
            l, r = i+1, n-1  # 设置左右指针
            while l < r:
                tmp = nums[i] + nums[l] + nums[r]  # 先计算三数和
                if tmp == target:  # 如果正好等于 target 就直接返回
                    return target
                elif tmp > target:  # 比目标大，右指针左移，使三数的和小一点
                    r -= 1
                    if abs(tmp-target) < diff:  # 更新最小差值
                        diff = abs(tmp-target)
                        res = tmp
                    while l < r and nums[r] == nums[r+1]:  # 如果左边的数还是一样的就接着向右移动
                        r -= 1    
                else:  # 比目标小，左指针右移，使三数的和大一点
                    l += 1
                    if abs(tmp-target) < diff:  # 更新最小差值
                        diff = abs(tmp-target)
                        res = tmp
                    while l < r and nums[l] == nums[l-1]:  # 如果右边的数还是一样的就接着向右移动
                        l += 1 
        return res  # 返回三数的和，而不是这三个数
