345. 反转字符串中的元音字母

Published: 周五 26 一月 2018
By Mega Hertz

In Leetcode.

自己写的辣鸡

class Solution(object):
    def reverseVowels(self, s):
        vowels = 'aeiouAEIOU'
        seq = []
        for c in s:
            if c in vowels:
                seq.append(c)
        res = []
        for c in s:
            if c in vowels:
                res.append(seq.pop())
            else:
                res.append(c)
        return ''.join(res)

双指针

时间复杂度: O(N) 空间复杂度: O(N)

class Solution:
    def reverseVowels(self, s):
        """
        :type s: str
        :rtype: str
        """
        vowels = 'aeiou'  # 这里用 set('a', 'e', 'i', 'o', 'u') 效率会更高, 因为 set 查找要比 list 快
        lst = list(s)
        l, r = 0, len(s) -1
        while l <= r:
            if lst[l].lower() not in vowels:
                l += 1
            elif lst[r].lower() not in vowels:
                r -= 1
            else:
                lst[l], lst[r] = lst[r], lst[l]
                l += 1
                r -= 1
        return ''.join(lst)

正则表达式

时间复杂度: O(N) 空间复杂度: O(1)

class Solution:
    def reverseVowels(self, s):
        vowels = re.findall(r'[aeiouAEIOU]', s)
        return re.sub('[aeiouAEIOU]', lambda m: vowels.pop(), s)

[aeiouAEIOU] 也可以换成 (?i)[aeiou]
